FILE(GLOB SRC *.cpp *.h res/logic_editor.rc2)

ENABLE_LANGUAGE(RC)

ADD_DEFINITIONS(-D_AFXDLL)
SET(CMAKE_MFC_FLAG 2) 
ADD_LIBRARY(logic_editor_dll SHARED ${SRC})

INCLUDE_DIRECTORIES(${LIBXML2_INCLUDE_DIR})
TARGET_LINK_LIBRARIES(logic_editor_dll ${LIBXML2_LIBRARIES} nelmisc nellogic nelnet)
SET_TARGET_PROPERTIES(logic_editor_dll PROPERTIES 
  VERSION ${NL_VERSION}
  DEBUG_POSTFIX "_d"
  RELEASE_POSTFIX "_r"
  PROJECT_LABEL "Tools, Logic: Logic Editor DLL")

IF(WIN32)
  SET_TARGET_PROPERTIES(logic_editor_dll PROPERTIES COMPILE_FLAGS "/YuStdAfx.h")
  SET_SOURCE_FILES_PROPERTIES(StdAfx.cpp PROPERTIES COMPILE_FLAGS "/YcStdAfx.h")
  SET_TARGET_PROPERTIES(logic_editor_dll PROPERTIES LINK_FLAGS "/NODEFAULTLIB:libcmt")
ENDIF(WIN32)


ADD_DEFINITIONS(${LIBXML2_DEFINITIONS})

INSTALL(TARGETS logic_editor_dll LIBRARY DESTINATION lib RUNTIME DESTINATION bin ARCHIVE DESTINATION lib COMPONENT toolsmisc)
